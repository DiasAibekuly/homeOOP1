CREATE TABLE guests (
                        guest_id VARCHAR(50) PRIMARY KEY,
                        full_name VARCHAR(100) NOT NULL
);

CREATE TABLE rooms (
                       room_number INT PRIMARY KEY,
                       price_per_night DECIMAL(10, 2) NOT NULL,
                       room_type VARCHAR(20) NOT NULL,
                       has_view BOOLEAN DEFAULT FALSE,
                       has_jacuzzi BOOLEAN DEFAULT FALSE
);

CREATE TABLE bookings (
                          booking_id SERIAL PRIMARY KEY,
                          room_number INT NOT NULL,
                          guest_id VARCHAR(50) NOT NULL,
                          nights INT NOT NULL,
                          total_cost DECIMAL(10, 2),
                          CONSTRAINT fk_room FOREIGN KEY (room_number) REFERENCES rooms(room_number),
                          CONSTRAINT fk_guest FOREIGN KEY (guest_id) REFERENCES guests(guest_id)
);
